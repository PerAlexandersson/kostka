\title{Non-real-rooted $h^*$-vectors of order polytopes\\from permutation posets}
\author{Per Alexandersson}
\address{Department of Mathematics, Stockholm University}
\email{per.alexandersson@math.su.se}

\begin{abstract}
We investigate the real-rootedness of $h^*$-polynomials of
order polytopes arising from permutation posets.
Starting from a permutation of Stembridge whose weak-order
descent polynomial fails real-rootedness, we observe that
the associated \emph{permutation poset} yields an order polytope
whose $h^*$-polynomial (equivalently, the $P$-Eulerian polynomial)
is also not real-rooted.
A computational search over 34{,}226 nearby 4321-avoiding
permutations finds exactly 22 non-real-rooted examples,
all of which are log-concave.
\end{abstract}

% =====================================================================
\section{Setup}\label{sec:setup}
% =====================================================================

Given a permutation $w = (w_1, \dotsc, w_n) \in S_n$, the
\defin{permutation poset} $P_w$ is the partial order on $\{1,\dotsc,n\}$
defined by
\[
  i <_{P_w} j \iff i < j \text{ and } w_i < w_j.
\]
This is always naturally labeled (every cover relation $i \lessdot j$
has $i < j$ as integers), so by Stanley's theorem~\cite[Thm.~4.5.14]{StanleyEC1}
the $h^*$-vector of the order polytope $\mathcal{O}(P_w)$ equals the
$P$-Eulerian polynomial:
\[
  h^*_k = \#\bigl\{\sigma \in \mathcal{L}(P_w) : \operatorname{des}(\sigma) = k\bigr\},
\]
where $\mathcal{L}(P_w)$ is the set of linear extensions and
$\operatorname{des}(\sigma) = \#\{i : \sigma(i) > \sigma(i+1)\}$.

If $w$ avoids the pattern $321$, then the longest decreasing
subsequence of $w$ has length ${\leq}\,2$, so by Dilworth's theorem
$P_w$ has width ${\leq}\,2$.  This makes both the frontier DP
(for Ehrhart computation) and linear extension enumeration
(for direct $P$-Eulerian computation) very efficient.


% =====================================================================
\section{Stembridge's permutation}\label{sec:stembridge}
% =====================================================================

Consider the 321-avoiding permutation on 17 elements:
\[
  w_0 = (2,4,6,8,10,1,12,3,15,5,17,7,9,11,13,14,16).
\]
This permutation was identified by Stembridge in connection with
weak-order descent polynomials.  The permutation poset $P_{w_0}$
has 25 cover relations, 7{,}016 linear extensions, and
$h^*$-polynomial
\[
  h^*(t) = 1 + 32t + 336t^2 + 1420t^3 + 2534t^4 + 1946t^5 + 658t^6 + 86t^7 + 3t^8.
\]
This polynomial is \textbf{log-concave} but \textbf{not real-rooted}.


% =====================================================================
\section{Neighborhood search}\label{sec:neighborhood}
% =====================================================================

Starting from $w_0$, we generate all permutations reachable by
at most $d$ arbitrary transpositions (not just adjacent), filter
to those avoiding the pattern $4321$, and test the $h^*$-polynomial
of each permutation poset for real-rootedness.

The $h^*$-polynomial is computed via the $P$-Eulerian polynomial
(streaming linear extension enumeration with descent counting,
using $O(n)$ memory).  For the 321-avoiding permutations in this
family, the number of linear extensions is moderate (${\leq}\,16{,}000$),
so this is fast (34{,}226 posets tested in 14 seconds using parallel
computation).

\begin{observation}\label{obs:rare}
Non-real-rootedness is rare: only 22 out of 34{,}226 permutations
fail, and the rate drops sharply with distance.
All 22 failures are log-concave.
Of the 22, exactly one contains the pattern~$321$ (at distance~3);
the remaining 21 are 321-avoiding.
\end{observation}


% =====================================================================
\section{The 22 non-real-rooted examples}\label{sec:table}
% =====================================================================

Table~\ref{tab:allnonRR} lists all 22 non-real-rooted $h^*$-polynomials
found in the search.
The polynomials separate into two families by degree:
11 of degree~8 and 11 of degree~7.

\begin{observation}\label{obs:twofamilies}
The 22 non-real-rooted $h^*$-polynomials have the following structure:
\begin{enumerate}
\item \textbf{Degree 8} (11 examples): the leading coefficient is
  always $3$, except for the unique 321-containing example which
  has leading coefficient $12$.
  The number of linear extensions ranges from $7{,}016$ to $8{,}359$
  (outlier: $15{,}388$ for the 321-containing case).
\item \textbf{Degree 7} (11 examples): the leading coefficient
  takes values $27, 30, 45, 54, 60$, with $45$ appearing 6 times.
  The number of linear extensions ranges from $4{,}883$ to $6{,}989$.
\item All 22 polynomials are \textbf{distinct}.
\item \textbf{All coefficients shift simultaneously} between
  Stembridge's original and each neighbor---the perturbation is
  global, not localized to a single coefficient.
\end{enumerate}
\end{observation}

\section{Questions}\label{sec:questions}
% =====================================================================

\begin{question}
Is the $h^*$-polynomial of $\mathcal{O}(P_w)$ always log-concave
for $w \in \operatorname{Av}(4321)$?
Our data is consistent with this.
\end{question}

\begin{question}
For which 321-avoiding permutations is $h^*(P_w; t)$ not real-rooted?
The 21 examples found here all lie within 3 transpositions of
Stembridge's $w_0$, suggesting a tight cluster.
Is there a combinatorial characterization?
\end{question}

\begin{question}
The leading coefficient of the degree-8 examples is always~$3$
(except for the unique 321-containing case, where it is~$12$).
Is there an explanation for this rigidity?
\end{question}
